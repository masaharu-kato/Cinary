//	sample_out.cpp
int main()
{
	Cinary::OBStream obs(std::ofstream("data.bin"));

	obs << 123 << 3.14159 << 'A' << "Hello, Cinary!";
	obs << std::vector<int>{85, 90, 45, 72, 93, 66, 36};
}


//	sample_in.cpp
int main()
{
	Cinary::IBStream ibs(std::ofstream("data.bin"));

	std::cout << (char)ibs << ", " << (double)ibs << ", " << (const char*)ibs << ", " << (int)ibs << "\n";
	for(auto v : std::vector<int>(ibs)) {
		std::cout << v << ", ";
	}
	std::cout << "\n";
}
